16"""An I/O event loop for non-blocking sockets. 

17 

18In Tornado 6.0, `.IOLoop` is a wrapper around the `asyncio` event 

19loop, with a slightly different interface for historical reasons. 

20Applications can use either the `.IOLoop` interface or the underlying 

21`asyncio` event loop directly (unless compatibility with older 

22versions of Tornado is desired, in which case `.IOLoop` must be used). 

23 

24Typical applications will use a single `IOLoop` object, accessed via 

25`IOLoop.current` class method. The `IOLoop.start` method (or 

26equivalently, `asyncio.AbstractEventLoop.run_forever`) should usually 

27be called at the end of the ``main()`` function. Atypical applications 

28may use more than one `IOLoop`, such as one `IOLoop` per thread, or 

29per `unittest` case. 

30 

31"""

88 import errno 

89 import functools 

90 import socket

92 import tornado.ioloop 

93 from tornado.iostream import IOStream 

94 

95 async def handle_connection(connection, address): 

96 stream = IOStream(connection) 

97 message = await stream.read_until_close() 

98 print("message from client:", message.decode().strip()) 

99 

100 def connection_ready(sock, fd, events): 

101 while True: 

102 try: 

103 connection, address = sock.accept() 

104 except socket.error as e: 

105 if e.args[0] not in (errno.EWOULDBLOCK, errno.EAGAIN): 

106 raise 

107 return 

108 connection.setblocking(0) 

109 io_loop = tornado.ioloop.IOLoop.current() 

110 io_loop.spawn_callback(handle_connection, connection, address)

114 sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) 

115 sock.setblocking(0) 

116 sock.bind(("", 8888)) 

117 sock.listen(128)

127 By default, a newly-constructed `IOLoop` becomes the thread's current 

128 `IOLoop`, unless there already is a current `IOLoop`. This behavior 

129 can be controlled with the ``make_current`` argument to the `IOLoop` 

130 constructor: if ``make_current=True``, the new `IOLoop` will always 

131 try to become current and it raises an error if there is already a 

132 current instance. If ``make_current=False``, the new `IOLoop` will 

133 not try to become current. 

134 

135 In general, an `IOLoop` cannot survive a fork or be shared across 

136 processes in any way. When multiple processes are being used, each 

137 process should create its own `IOLoop`, which also implies that 

138 any objects which depend on the `IOLoop` (such as 

139 `.AsyncHTTPClient`) must also be created in the child processes. 

140 As a guideline, anything that starts processes (including the 

141 `tornado.process` and `multiprocessing` modules) should do so as 

142 early as possible, ideally the first thing the application does 

143 after loading its configuration in ``main()``. 

144 

145 .. versionchanged:: 4.2 

146 Added the ``make_current`` keyword argument to the `IOLoop` 

147 constructor. 

148 

149 .. versionchanged:: 5.0 

150 

151 Uses the `asyncio` event loop by default. The 

152 ``IOLoop.configure`` method cannot be used on Python 3 except 

153 to redundantly specify the `asyncio` event loop.

343 def close(self, all_fds: bool = False) -> None: 

344 """Closes the `IOLoop`, freeing any resources used. 

345 

346 If ``all_fds`` is true, all file descriptors registered on the 

347 IOLoop will be closed (not just the ones created by the 

348 `IOLoop` itself). 

349 

350 Many applications will only use a single `IOLoop` that runs for the 

351 entire lifetime of the process. In that case closing the `IOLoop` 

352 is not necessary since everything will be cleaned up when the 

353 process exits. `IOLoop.close` is provided mainly for scenarios 

354 such as unit tests, which create and destroy a large number of 

355 ``IOLoops``. 

356 

357 An `IOLoop` must be completely stopped before it can be closed. This 

358 means that `IOLoop.stop()` must be called *and* `IOLoop.start()` must 

359 be allowed to return before attempting to call `IOLoop.close()`. 

360 Therefore the call to `close` will usually appear just after 

361 the call to `start` rather than near the call to `stop`. 

362 

363 .. versionchanged:: 3.1 

364 If the `IOLoop` implementation supports non-integer objects 

365 for "file descriptors", those objects will have their 

366 ``close`` method when ``all_fds`` is true. 

367 """

460 def run_sync(self, func: Callable, timeout: float = None) -> Any: 

461 """Starts the `IOLoop`, runs the given function, and stops the loop. 

462 

463 The function must return either an awaitable object or 

464 ``None``. If the function returns an awaitable object, the 

465 `IOLoop` will run until the awaitable is resolved (and 

466 `run_sync()` will return the awaitable's result). If it raises 

467 an exception, the `IOLoop` will stop and the exception will be 

468 re-raised to the caller. 

469 

470 The keyword-only argument ``timeout`` may be used to set 

471 a maximum duration for the function. If the timeout expires, 

472 a `tornado.util.TimeoutError` is raised. 

473 

474 This method is useful to allow asynchronous calls in a 

475 ``main()`` function:: 

476 

477 async def main(): 

478 # do stuff... 

479 

480 if __name__ == '__main__': 

481 IOLoop.current().run_sync(main) 

482 

483 .. versionchanged:: 4.3 

484 Returning a non-``None``, non-awaitable value is now an error. 

485 

486 .. versionchanged:: 5.0 

487 If a timeout occurs, the ``func`` coroutine will be cancelled. 

488 

489 """ 

490 future_cell = [None] # type: List[Optional[Future]] 

491 

492 def run() -> None: 

493 try: 

494 result = func() 

495 if result is not None: 

496 from tornado.gen import convert_yielded 

497 

498 result = convert_yielded(result) 

499 except Exception: 

500 fut = Future() # type: Future[Any] 

501 future_cell[0] = fut 

502 future_set_exc_info(fut, sys.exc_info()) 

503 else: 

504 if is_future(result): 

505 future_cell[0] = result 

506 else: 

507 fut = Future() 

508 future_cell[0] = fut 

509 fut.set_result(result) 

510 assert future_cell[0] is not None 

511 self.add_future(future_cell[0], lambda future: self.stop()) 

512 

513 self.add_callback(run) 

514 if timeout is not None: 

515 

516 def timeout_callback() -> None: 

517 # If we can cancel the future, do so and wait on it. If not, 

518 # Just stop the loop and return with the task still pending. 

519 # (If we neither cancel nor wait for the task, a warning 

520 # will be logged). 

521 assert future_cell[0] is not None 

522 if not future_cell[0].cancel(): 

523 self.stop() 

524 

525 timeout_handle = self.add_timeout(self.time() + timeout, timeout_callback) 

526 self.start() 

527 if timeout is not None: 

528 self.remove_timeout(timeout_handle) 

529 assert future_cell[0] is not None 

530 if future_cell[0].cancelled() or not future_cell[0].done(): 

531 raise TimeoutError("Operation timed out after %s seconds" % timeout) 

532 return future_cell[0].result()

548 def add_timeout( 

549 self, 

550 deadline: Union[float, datetime.timedelta], 

551 callback: Callable[..., None], 

552 *args: Any, 

553 **kwargs: Any 

554 ) -> object: 

555 """Runs the ``callback`` at the time ``deadline`` from the I/O loop. 

556 

557 Returns an opaque handle that may be passed to 

558 `remove_timeout` to cancel. 

559 

560 ``deadline`` may be a number denoting a time (on the same 

561 scale as `IOLoop.time`, normally `time.time`), or a 

562 `datetime.timedelta` object for a deadline relative to the 

563 current time. Since Tornado 4.0, `call_later` is a more 

564 convenient alternative for the relative case since it does not 

565 require a timedelta object. 

566 

567 Note that it is not safe to call `add_timeout` from other threads. 

568 Instead, you must use `add_callback` to transfer control to the 

569 `IOLoop`'s thread, and then call `add_timeout` from there. 

570 

571 Subclasses of IOLoop must implement either `add_timeout` or 

572 `call_at`; the default implementations of each will call 

573 the other. `call_at` is usually easier to implement, but 

574 subclasses that wish to maintain compatibility with Tornado 

575 versions prior to 4.0 must use `add_timeout` instead. 

576 

577 .. versionchanged:: 4.0 

578 Now passes through ``*args`` and ``**kwargs`` to the callback. 

579 """ 

580 if isinstance(deadline, numbers.Real): 

581 return self.call_at(deadline, callback, *args, **kwargs) 

582 elif isinstance(deadline, datetime.timedelta): 

583 return self.call_at( 

584 self.time() + deadline.total_seconds(), callback, *args, **kwargs 

585 ) 

586 else: 

587 raise TypeError("Unsupported deadline %r" % deadline)

677 awaitables (unlike most of Tornado where the two are 

678 interchangeable). 

679 """ 

680 if isinstance(future, Future): 

681 # Note that we specifically do not want the inline behavior of 

682 # tornado.concurrent.future_add_done_callback. We always want 

683 # this callback scheduled on the next IOLoop iteration (which 

684 # asyncio.Future always does). 

685 # 

686 # Wrap the callback in self._run_callback so we control 

687 # the error logging (i.e. it goes to tornado.log.app_log 

688 # instead of asyncio's log). 

689 future.add_done_callback( 

690 lambda f: self._run_callback(functools.partial(callback, future)) 

691 ) 

692 else: 

693 assert is_future(future) 

694 # For concurrent futures, we use self.add_callback, so 

695 # it's fine if future_add_done_callback inlines that call. 

696 future_add_done_callback( 

697 future, lambda f: self.add_callback(callback, future) 

698 )

700 def run_in_executor( 

701 self, 

702 executor: Optional[concurrent.futures.Executor], 

703 func: Callable[..., _T], 

704 *args: Any 

705 ) -> Awaitable[_T]: 

706 """Runs a function in a ``concurrent.futures.Executor``. If 

707 ``executor`` is ``None``, the IO loop's default executor will be used. 

708 

709 Use `functools.partial` to pass keyword arguments to ``func``. 

710 

711 .. versionadded:: 5.0 

712 """ 

713 if executor is None: 

714 if not hasattr(self, "_executor"): 

715 from tornado.process import cpu_count 

716 

717 self._executor = concurrent.futures.ThreadPoolExecutor( 

718 max_workers=(cpu_count() * 5) 

719 ) # type: concurrent.futures.Executor 

720 executor = self._executor 

721 c_future = executor.submit(func, *args) 

722 # Concurrent Futures are not usable with await. Wrap this in a 

723 # Tornado Future instead, using self.add_future for thread-safety. 

724 t_future = Future() # type: Future[_T] 

725 self.add_future(c_future, lambda f: chain_future(f, t_future)) 

726 return t_future

728 def set_default_executor(self, executor: concurrent.futures.Executor) -> None: 

729 """Sets the default executor to use with :meth:`run_in_executor`. 

730 

731 .. versionadded:: 5.0 

732 """ 

733 self._executor = executor

735 def _run_callback(self, callback: Callable[[], Any]) -> None: 

736 """Runs a callback with error handling. 

737 

738 .. versionchanged:: 6.0 

739 

740 CancelledErrors are no longer logged. 

741 """ 

742 try: 

743 ret = callback() 

744 if ret is not None: 

745 from tornado import gen 

746 

747 # Functions that return Futures typically swallow all 

748 # exceptions and store them in the Future. If a Future 

749 # makes it out to the IOLoop, ensure its exception (if any) 

750 # gets logged too. 

751 try: 

752 ret = gen.convert_yielded(ret) 

753 except gen.BadYieldError: 

754 # It's not unusual for add_callback to be used with 

755 # methods returning a non-None and non-yieldable 

756 # result, which should just be ignored. 

757 pass 

758 else: 

759 self.add_future(ret, self._discard_future_result) 

760 except asyncio.CancelledError: 

761 pass 

762 except Exception: 

763 app_log.error("Exception in callback %r", callback, exc_info=True)

765 def _discard_future_result(self, future: Future) -> None: 

766 """Avoid unhandled-exception warnings from spawned coroutines.""" 

767 future.result()

771 ) -> Tuple[int, Union[int, _Selectable]]: 

772 # """Returns an (fd, obj) pair from an ``fd`` parameter. 

773 

774 # We accept both raw file descriptors and file-like objects as 

775 # input to `add_handler` and related methods. When a file-like 

776 # object is passed, we must retain the object itself so we can 

777 # close it correctly when the `IOLoop` shuts down, but the

780 # always return the descriptor itself). 

781

783 # not generally be used by application code. 

784 

785 # .. versionadded:: 4.0 

786 # """ 

787 if isinstance(fd, int): 

788 return fd, fd 

789 return fd.fileno(), fd

812class _Timeout(object): 

813 """An IOLoop timeout, a UNIX timestamp and a callback""" 

814 

815 # Reduce memory overhead when there are lots of pending callbacks 

816 __slots__ = ["deadline", "callback", "tdeadline"] 

817 

818 def __init__( 

819 self, deadline: float, callback: Callable[[], None], io_loop: IOLoop 

820 ) -> None: 

821 if not isinstance(deadline, numbers.Real): 

822 raise TypeError("Unsupported deadline %r" % deadline) 

823 self.deadline = deadline 

824 self.callback = callback 

825 self.tdeadline = ( 

826 deadline, 

827 next(io_loop._timeout_counter), 

828 ) # type: Tuple[float, int] 

829 

830 # Comparison methods to sort by deadline, with object id as a tiebreaker 

831 # to guarantee a consistent ordering. The heapq module uses __le__ 

832 # in python2.5, and __lt__ in 2.6+ (sort() and most other comparisons 

833 # use __lt__). 

834 def __lt__(self, other: "_Timeout") -> bool: 

835 return self.tdeadline < other.tdeadline 

836 

837 def __le__(self, other: "_Timeout") -> bool: 

838 return self.tdeadline <= other.tdeadline

841class PeriodicCallback(object): 

842 """Schedules the given callback to be called periodically. 

843 

844 The callback is called every ``callback_time`` milliseconds. 

845 Note that the timeout is given in milliseconds, while most other 

846 time-related functions in Tornado use seconds. 

847 

848 If ``jitter`` is specified, each callback time will be randomly selected 

849 within a window of ``jitter * callback_time`` milliseconds. 

850 Jitter can be used to reduce alignment of events with similar periods. 

851 A jitter of 0.1 means allowing a 10% variation in callback time. 

852 The window is centered on ``callback_time`` so the total number of calls 

853 within a given interval should not be significantly affected by adding 

854 jitter. 

855 

856 If the callback runs for longer than ``callback_time`` milliseconds, 

857 subsequent invocations will be skipped to get back on schedule. 

858 

859 `start` must be called after the `PeriodicCallback` is created. 

860 

861 .. versionchanged:: 5.0 

862 The ``io_loop`` argument (deprecated since version 4.1) has been removed. 

863 

864 .. versionchanged:: 5.1 

865 The ``jitter`` argument is added. 

866 """ 

867 

868 def __init__( 

869 self, callback: Callable[[], None], callback_time: float, jitter: float = 0 

870 ) -> None: 

871 self.callback = callback 

872 if callback_time <= 0: 

873 raise ValueError("Periodic callback must have a positive callback_time") 

874 self.callback_time = callback_time 

875 self.jitter = jitter 

876 self._running = False 

877 self._timeout = None # type: object 

878 

879 def start(self) -> None: 

880 """Starts the timer.""" 

881 # Looking up the IOLoop here allows to first instantiate the 

882 # PeriodicCallback in another thread, then start it using 

883 # IOLoop.add_callback(). 

884 self.io_loop = IOLoop.current() 

885 self._running = True 

886 self._next_timeout = self.io_loop.time() 

887 self._schedule_next() 

888 

889 def stop(self) -> None: 

890 """Stops the timer.""" 

891 self._running = False 

892 if self._timeout is not None: 

893 self.io_loop.remove_timeout(self._timeout) 

894 self._timeout = None 

895 

896 def is_running(self) -> bool: 

897 """Returns ``True`` if this `.PeriodicCallback` has been started. 

898 

899 .. versionadded:: 4.1 

900 """ 

901 return self._running 

902 

903 def _run(self) -> None: 

904 if not self._running: 

905 return 

906 try: 

907 return self.callback() 

908 except Exception: 

909 app_log.error("Exception in callback %r", self.callback, exc_info=True) 

910 finally: 

911 self._schedule_next() 

912 

913 def _schedule_next(self) -> None: 

914 if self._running: 

915 self._update_next(self.io_loop.time()) 

916 self._timeout = self.io_loop.add_timeout(self._next_timeout, self._run) 

917 

918 def _update_next(self, current_time: float) -> None: 

919 callback_time_sec = self.callback_time / 1000.0 

920 if self.jitter: 

921 # apply jitter fraction 

922 callback_time_sec *= 1 + (self.jitter * (random.random() - 0.5)) 

923 if self._next_timeout <= current_time: 

924 # The period should be measured from the start of one call 

925 # to the start of the next. If one call takes too long, 

926 # skip cycles to get back to a multiple of the original 

927 # schedule. 

928 self._next_timeout += ( 

929 math.floor((current_time - self._next_timeout) / callback_time_sec) + 1 

930 ) * callback_time_sec 

931 else: 

932 # If the clock moved backwards, ensure we advance the next 

933 # timeout instead of recomputing the same value again. 

934 # This may result in long gaps between callbacks if the 

935 # clock jumps backwards by a lot, but the far more common 

936 # scenario is a small NTP adjustment that should just be 

937 # ignored. 

938 # 

939 # Note that on some systems if time.time() runs slower 

940 # than time.monotonic() (most common on windows), we 

941 # effectively experience a small backwards time jump on 

942 # every iteration because PeriodicCallback uses 

943 # time.time() while asyncio schedules callbacks using 

944 # time.monotonic(). 

945 # https://github.com/tornadoweb/tornado/issues/2333 

946 self._next_timeout += callback_time_sec
