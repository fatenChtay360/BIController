116class AsyncTestCase(unittest.TestCase): 

117 """`~unittest.TestCase` subclass for testing `.IOLoop`-based 

118 asynchronous code. 

119 

120 The unittest framework is synchronous, so the test must be 

121 complete by the time the test method returns. This means that 

122 asynchronous code cannot be used in quite the same way as usual 

123 and must be adapted to fit. To write your tests with coroutines, 

124 decorate your test methods with `tornado.testing.gen_test` instead 

125 of `tornado.gen.coroutine`. 

126 

127 This class also provides the (deprecated) `stop()` and `wait()` 

128 methods for a more manual style of testing. The test method itself 

129 must call ``self.wait()``, and asynchronous callbacks should call 

130 ``self.stop()`` to signal completion. 

131 

132 By default, a new `.IOLoop` is constructed for each test and is available 

133 as ``self.io_loop``. If the code being tested requires a 

134 global `.IOLoop`, subclasses should override `get_new_ioloop` to return it. 

135 

136 The `.IOLoop`'s ``start`` and ``stop`` methods should not be 

137 called directly. Instead, use `self.stop <stop>` and `self.wait 

138 <wait>`. Arguments passed to ``self.stop`` are returned from 

139 ``self.wait``. It is possible to have multiple ``wait``/``stop`` 

140 cycles in the same test. 

141 

142 Example:: 

143 

144 # This test uses coroutine style. 

145 class MyTestCase(AsyncTestCase): 

146 @tornado.testing.gen_test 

147 def test_http_fetch(self): 

148 client = AsyncHTTPClient() 

149 response = yield client.fetch("http://www.tornadoweb.org") 

150 # Test contents of response 

151 self.assertIn("FriendFeed", response.body) 

152 

153 # This test uses argument passing between self.stop and self.wait. 

154 class MyTestCase2(AsyncTestCase): 

155 def test_http_fetch(self): 

156 client = AsyncHTTPClient() 

157 client.fetch("http://www.tornadoweb.org/", self.stop) 

158 response = self.wait() 

159 # Test contents of response 

160 self.assertIn("FriendFeed", response.body) 

161 """

163 def __init__(self, methodName: str = "runTest") -> None:

173 # the generator. Replace the test method with a wrapper that will 

174 # make sure it's not an undecorated generator. 

175 setattr(self, methodName, _TestMethodWrapper(getattr(self, methodName)))

185 def tearDown(self) -> None: 

186 # Native coroutines tend to produce warnings if they're not 

187 # allowed to run to completion. It's difficult to ensure that 

188 # this always happens in tests, so cancel any tasks that are 

189 # still pending by the time we get here. 

190 asyncio_loop = self.io_loop.asyncio_loop # type: ignore 

191 if hasattr(asyncio, "all_tasks"): # py37 

192 tasks = asyncio.all_tasks(asyncio_loop) # type: ignore 

193 else: 

194 tasks = asyncio.Task.all_tasks(asyncio_loop) 

195 # Tasks that are done may still appear here and may contain 

196 # non-cancellation exceptions, so filter them out. 

197 tasks = [t for t in tasks if not t.done()] 

198 for t in tasks: 

199 t.cancel() 

200 # Allow the tasks to run and finalize themselves (which means 

201 # raising a CancelledError inside the coroutine). This may 

202 # just transform the "task was destroyed but it is pending" 

203 # warning into a "uncaught CancelledError" warning, but 

204 # catching CancelledErrors in coroutines that may leak is 

205 # simpler than ensuring that no coroutines leak. 

206 if tasks: 

207 done, pending = self.io_loop.run_sync(lambda: asyncio.wait(tasks)) 

208 assert not pending 

209 # If any task failed with anything but a CancelledError, raise it. 

210 for f in done: 

211 try: 

212 f.result() 

213 except asyncio.CancelledError: 

214 pass 

215 

216 # Clean up Subprocess, so it can be used again with a new ioloop. 

217 Subprocess.uninitialize() 

218 self.io_loop.clear_current() 

219 if not isinstance(self.io_loop, _NON_OWNED_IOLOOPS): 

220 # Try to clean up any file descriptors left open in the ioloop. 

221 # This avoids leaks, especially when tests are run repeatedly 

222 # in the same process with autoreload (because curl does not 

223 # set FD_CLOEXEC on its file descriptors) 

224 self.io_loop.close(all_fds=True) 

225 super(AsyncTestCase, self).tearDown() 

226 # In case an exception escaped or the StackContext caught an exception 

227 # when there wasn't a wait() to re-raise it, do so here. 

228 # This is our last chance to raise an exception in a way that the 

229 # unittest machinery understands. 

230 self.__rethrow()

273 """Stops the `.IOLoop`, causing one pending (or future) call to `wait()` 

274 to return. 

275 

276 Keyword arguments or a single positional argument passed to `stop()` are 

277 saved and will be returned by `wait()`.

280 

281 `stop` and `wait` are deprecated; use ``@gen_test`` instead.

283 assert _arg is None or not kwargs 

284 self.__stop_args = kwargs or _arg

290 def wait( 

291 self, condition: Callable[..., bool] = None, timeout: float = None 

292 ) -> None: 

293 """Runs the `.IOLoop` until stop is called or timeout has passed. 

294 

295 In the event of a timeout, an exception will be thrown. The 

296 default timeout is 5 seconds; it may be overridden with a 

297 ``timeout`` keyword argument or globally with the 

298 ``ASYNC_TEST_TIMEOUT`` environment variable. 

299 

300 If ``condition`` is not ``None``, the `.IOLoop` will be restarted 

301 after `stop()` until ``condition()`` returns ``True``. 

302 

303 .. versionchanged:: 3.1 

304 Added the ``ASYNC_TEST_TIMEOUT`` environment variable. 

305 

306 .. deprecated:: 5.1 

307 

308 `stop` and `wait` are deprecated; use ``@gen_test`` instead. 

309 """ 

310 if timeout is None: 

311 timeout = get_async_test_timeout() 

312 

313 if not self.__stopped: 

314 if timeout: 

315 

316 def timeout_func() -> None: 

317 try: 

318 raise self.failureException( 

319 "Async operation timed out after %s seconds" % timeout 

320 ) 

321 except Exception: 

322 self.__failure = sys.exc_info() 

323 self.stop() 

324 

325 self.__timeout = self.io_loop.add_timeout( 

326 self.io_loop.time() + timeout, timeout_func 

327 ) 

328 while True: 

329 self.__running = True 

330 self.io_loop.start() 

331 if self.__failure is not None or condition is None or condition(): 

332 break 

333 if self.__timeout is not None: 

334 self.io_loop.remove_timeout(self.__timeout) 

335 self.__timeout = None 

336 assert self.__stopped 

337 self.__stopped = False 

338 self.__rethrow() 

339 result = self.__stop_args 

340 self.__stop_args = None 

341 return result

344class AsyncHTTPTestCase(AsyncTestCase): 

345 """A test case that starts up an HTTP server. 

346 

347 Subclasses must override `get_app()`, which returns the 

348 `tornado.web.Application` (or other `.HTTPServer` callback) to be tested. 

349 Tests will typically use the provided ``self.http_client`` to fetch 

350 URLs from this server. 

351 

352 Example, assuming the "Hello, world" example from the user guide is in 

353 ``hello.py``:: 

354 

355 import hello 

356 

357 class TestHelloApp(AsyncHTTPTestCase): 

358 def get_app(self): 

359 return hello.make_app() 

360 

361 def test_homepage(self): 

362 response = self.fetch('/') 

363 self.assertEqual(response.code, 200) 

364 self.assertEqual(response.body, 'Hello, world') 

365 

366 That call to ``self.fetch()`` is equivalent to :: 

367 

368 self.http_client.fetch(self.get_url('/'), self.stop) 

369 response = self.wait() 

370 

371 which illustrates how AsyncTestCase can turn an asynchronous operation, 

372 like ``http_client.fetch()``, into a synchronous operation. If you need 

373 to do other asynchronous operations in tests, you'll probably need to use 

374 ``stop()`` and ``wait()`` yourself. 

375 """

399 def fetch( 

400 self, path: str, raise_error: bool = False, **kwargs: Any 

401 ) -> HTTPResponse: 

402 """Convenience method to synchronously fetch a URL. 

403 

404 The given path will be appended to the local server's host and 

405 port. Any additional keyword arguments will be passed directly to 

406 `.AsyncHTTPClient.fetch` (and so could be used to pass 

407 ``method="POST"``, ``body="..."``, etc). 

408 

409 If the path begins with http:// or https://, it will be treated as a 

410 full URL and will be fetched as-is. 

411 

412 If ``raise_error`` is ``True``, a `tornado.httpclient.HTTPError` will 

413 be raised if the response code is not 200. This is the same behavior 

414 as the ``raise_error`` argument to `.AsyncHTTPClient.fetch`, but 

415 the default is ``False`` here (it's ``True`` in `.AsyncHTTPClient`) 

416 because tests often need to deal with non-200 response codes. 

417 

418 .. versionchanged:: 5.0 

419 Added support for absolute URLs. 

420 

421 .. versionchanged:: 5.1 

422 

423 Added the ``raise_error`` argument. 

424 

425 .. deprecated:: 5.1 

426 

427 This method currently turns any exception into an 

428 `.HTTPResponse` with status code 599. In Tornado 6.0, 

429 errors other than `tornado.httpclient.HTTPError` will be 

430 passed through, and ``raise_error=False`` will only 

431 suppress errors that would be raised due to non-200 

432 response codes.

437 else: 

438 url = self.get_url(path)

441 timeout=get_async_test_timeout(), 

442 )

527 """Testing equivalent of ``@gen.coroutine``, to be applied to test methods. 

528 

529 ``@gen.coroutine`` cannot be used on tests because the `.IOLoop` is not

541 overridden globally with the ``ASYNC_TEST_TIMEOUT`` environment variable, 

542 or for each test with the ``timeout`` keyword argument::

545 @gen_test(timeout=10) 

546 def test_something_slow(self): 

547 response = yield self.http_client.fetch(self.get_url('/')) 

548

550 `AsyncTestCase.wait`, and `AsyncHTTPTestCase.fetch`. Use ``yield 

551 self.http_client.fetch(self.get_url())`` as shown above instead. 

552 

553 .. versionadded:: 3.1 

554 The ``timeout`` argument and ``ASYNC_TEST_TIMEOUT`` environment 

555 variable. 

556 

557 .. versionchanged:: 4.0 

558 The wrapper now passes along ``*args, **kwargs`` so it can be used 

559 on functions with arguments. 

560

562 if timeout is None:

565 def wrap(f: Callable[..., Union[Generator, "Coroutine"]]) -> Callable[..., None]: 

566 # Stack up several decorators to allow us to access the generator 

567 # object itself. In the innermost wrapper, we capture the generator 

568 # and save it in an attribute of self. Next, we run the wrapped 

569 # function through @gen.coroutine. Finally, the coroutine is 

570 # wrapped again to make it synchronous with run_sync. 

571 # 

572 # This is a good case study arguing for either some sort of 

573 # extensibility in the gen decorators or cancellation support. 

574 @functools.wraps(f) 

575 def pre_coroutine(self, *args, **kwargs): 

576 # type: (AsyncTestCase, *Any, **Any) -> Union[Generator, Coroutine] 

577 # Type comments used to avoid pypy3 bug. 

578 result = f(self, *args, **kwargs) 

579 if isinstance(result, Generator) or inspect.iscoroutine(result): 

580 self._test_generator = result 

581 else: 

582 self._test_generator = None 

583 return result 

584 

585 if inspect.iscoroutinefunction(f): 

586 coro = pre_coroutine 

587 else: 

588 coro = gen.coroutine(pre_coroutine) 

589 

590 @functools.wraps(coro) 

591 def post_coroutine(self, *args, **kwargs): 

592 # type: (AsyncTestCase, *Any, **Any) -> None 

593 try: 

594 return self.io_loop.run_sync( 

595 functools.partial(coro, self, *args, **kwargs), timeout=timeout 

596 ) 

597 except TimeoutError as e: 

598 # run_sync raises an error with an unhelpful traceback. 

599 # If the underlying generator is still running, we can throw the 

600 # exception back into it so the stack trace is replaced by the 

601 # point where the test is stopped. The only reason the generator 

602 # would not be running would be if it were cancelled, which means 

603 # a native coroutine, so we can rely on the cr_running attribute. 

604 if self._test_generator is not None and getattr( 

605 self._test_generator, "cr_running", True 

606 ): 

607 self._test_generator.throw(type(e), e) 

608 # In case the test contains an overly broad except 

609 # clause, we may get back here. 

610 # Coroutine was stopped or didn't raise a useful stack trace, 

611 # so re-raise the original exception which is better than nothing. 

612 raise 

613 

614 return post_coroutine 

615 

616 if func is not None: 

617 # Used like: 

618 # @gen_test 

619 # def f(self): 

620 # pass 

621 return wrap(func) 

622 else: 

623 # Used like @gen_test(timeout=10) 

624 return wrap

652 logger: Union[logging.Logger, basestring_type], 

653 regex: str, 

654 required: bool = True, 

655 ) -> None: 

656 """Constructs an ExpectLog context manager. 

657 

658 :param logger: Logger object (or name of logger) to watch. Pass 

659 an empty string to watch the root logger. 

660 :param regex: Regular expression to match. Any log entries on 

661 the specified logger that match this regex will be suppressed. 

662 :param required: If true, an exception will be raised if the end of 

663 the ``with`` statement is reached without matching any log entries. 

664 """ 

665 if isinstance(logger, basestring_type): 

666 logger = logging.getLogger(logger) 

667 self.logger = logger 

668 self.regex = re.compile(regex) 

669 self.required = required 

670 self.matched = False 

671 self.logged_stack = False 

672 

673 def filter(self, record: logging.LogRecord) -> bool: 

674 if record.exc_info: 

675 self.logged_stack = True 

676 message = record.getMessage() 

677 if self.regex.match(message): 

678 self.matched = True 

679 return False 

680 return True 

681 

682 def __enter__(self) -> "ExpectLog": 

683 self.logger.addFilter(self) 

684 return self 

685 

686 def __exit__( 

687 self, 

688 typ: "Optional[Type[BaseException]]", 

689 value: Optional[BaseException], 

690 tb: Optional[TracebackType], 

691 ) -> None: 

692 self.logger.removeFilter(self) 

693 if not typ and self.required and not self.matched: 

694 raise Exception("did not get expected log message") 

695

697def main(**kwargs: Any) -> None: 

698 """A simple test runner. 

699 

700 This test runner is essentially equivalent to `unittest.main` from 

701 the standard library, but adds support for Tornado-style option 

702 parsing and log formatting. It is *not* necessary to use this 

703 `main` function to run tests using `AsyncTestCase`; these tests 

704 are self-contained and can run with any test runner. 

705 

706 The easiest way to run a test is via the command line:: 

707 

708 python -m tornado.testing tornado.test.web_test 

709 

710 See the standard library ``unittest`` module for ways in which 

711 tests can be specified. 

712 

713 Projects with many tests may wish to define a test script like 

714 ``tornado/test/runtests.py``. This script should define a method 

715 ``all()`` which returns a test suite and then call 

716 `tornado.testing.main()`. Note that even when a test script is 

717 used, the ``all()`` test suite may be overridden by naming a 

718 single test on the command line:: 

719 

720 # Runs all tests 

721 python -m tornado.test.runtests 

722 # Runs one test 

723 python -m tornado.test.runtests tornado.test.web_test 

724 

725 Additional keyword arguments passed through to ``unittest.main()``. 

726 For example, use ``tornado.testing.main(verbosity=2)`` 

727 to show many test details as they are run. 

728 See http://docs.python.org/library/unittest.html#unittest.main 

729 for full argument list. 

730 

731 .. versionchanged:: 5.0 

732 

733 This function produces no output of its own; only that produced 

734 by the `unittest` module (previously it would add a PASS or FAIL 

735 log message). 

736 """ 

737 from tornado.options import define, options, parse_command_line 

738 

739 define( 

740 "exception_on_interrupt", 

741 type=bool, 

742 default=True, 

743 help=( 

744 "If true (default), ctrl-c raises a KeyboardInterrupt " 

745 "exception. This prints a stack trace but cannot interrupt " 

746 "certain operations. If false, the process is more reliably " 

747 "killed, but does not print a stack trace." 

748 ), 

749 ) 

750 

751 # support the same options as unittest's command-line interface 

752 define("verbose", type=bool) 

753 define("quiet", type=bool) 

754 define("failfast", type=bool) 

755 define("catch", type=bool) 

756 define("buffer", type=bool) 

757 

758 argv = [sys.argv[0]] + parse_command_line(sys.argv) 

759 

760 if not options.exception_on_interrupt: 

761 signal.signal(signal.SIGINT, signal.SIG_DFL) 

762 

763 if options.verbose is not None: 

764 kwargs["verbosity"] = 2 

765 if options.quiet is not None: 

766 kwargs["verbosity"] = 0 

767 if options.failfast is not None: 

768 kwargs["failfast"] = True 

769 if options.catch is not None: 

770 kwargs["catchbreak"] = True 

771 if options.buffer is not None: 

772 kwargs["buffer"] = True 

773 

774 if __name__ == "__main__" and len(argv) == 1: 

775 print("No tests specified", file=sys.stderr) 

776 sys.exit(1) 

777 # In order to be able to run tests by their fully-qualified name 

778 # on the command line without importing all tests here, 

779 # module must be set to None. Python 3.2's unittest.main ignores 

780 # defaultTest if no module is given (it tries to do its own 

781 # test discovery, which is incompatible with auto2to3), so don't 

782 # set module if we're not asking for a specific test. 

783 if len(argv) > 1: 

784 unittest.main(module=None, argv=argv, **kwargs) # type: ignore 

785 else: 

786 unittest.main(defaultTest="all", argv=argv, **kwargs) 

787 

788 

789if __name__ == "__main__": 

790 main()
