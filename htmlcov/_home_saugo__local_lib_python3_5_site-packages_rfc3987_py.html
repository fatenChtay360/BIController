19r""" 

20Parsing and validation of URIs (RFC 3986) and IRIs (RFC 3987). 

21 

22This module provides regular expressions according to `RFC 3986 "Uniform 

23Resource Identifier (URI): Generic Syntax" 

24<http://tools.ietf.org/html/rfc3986>`_ and `RFC 3987 "Internationalized 

25Resource Identifiers (IRIs)" <http://tools.ietf.org/html/rfc3987>`_, and 

26utilities for composition and relative resolution of references. 

27 

28 

29API 

30--- 

31 

32**match** (string, rule='IRI_reference') 

33 {match.__doc__} 

34 

35**parse** (string, rule='IRI_reference') 

36 {parse.__doc__} 

37 

38**compose** (\*\*parts) 

39 {compose.__doc__} 

40 

41**resolve** (base, uriref, strict=True, return_parts=False) 

42 {resolve.__doc__} 

43 

44**patterns** 

45 A dict of regular expressions with useful group names. 

46 Compilable (with regex_ only) without need for any particular compilation 

47 flag. 

48 

49**[bmp_][u]patterns[_no_names]** 

50 Alternative versions of `patterns`. 

51 [u]nicode strings without group names for the re_ module. 

52 BMP only for narrow builds. 

53 

54**get_compiled_pattern** (rule, flags=0) 

55 {get_compiled_pattern.__doc__} 

56 

57**format_patterns** (\*\*names) 

58 {format_patterns.__doc__} 

59 

60 

61Dependencies

64Some features require regex_.

289def format_patterns(**names): 

290 r"""Returns a dict of patterns (regular expressions) keyed by 

291 `rule names for URIs`_ and `rule names for IRIs`_. 

292 

293 See also the module level dicts of patterns, and `get_compiled_pattern`. 

294 

295 To wrap a rule in a named capture group, pass it as keyword argument: 

296 rule_name='group_name'. By default, the formatted patterns contain no 

297 named groups. 

298 

299 Patterns are `str` instances (be it in python 2.x or 3.x) containing ASCII 

300 characters only. 

301 

302 Caveats: 

303 

304 - with re_, named capture groups cannot occur on multiple branches of an 

305 alternation 

306 

307 - with re_ before python 3.3, ``\u`` and ``\U`` escapes must be 

308 preprocessed (see `issue3665 <http://bugs.python.org/issue3665>`_) 

309 

310 - on narrow builds, character ranges beyond BMP are not supported 

311 

312 .. _rule names for URIs: http://tools.ietf.org/html/rfc3986#appendix-A 

313 .. _rule names for IRIs: http://tools.ietf.org/html/rfc3987#section-2.2 

314 """ 

315 formatted = {} 

316 for name, pat in _common_rules[::-1] + _uri_rules[::-1] + _iri_rules[::-1]: 

317 if name in names: 

318 n = names[name] 

319 if callable(n): 

320 pat = n(pat) 

321 else: 

322 pat = '(?P<%s>%s)' % (n, pat) 

323 formatted[name] = pat.format(**formatted) 

324 return formatted

369def get_compiled_pattern(rule, flags=0): 

370 """Returns a compiled pattern object for a rule name or template string. 

371 

372 Usage for validation:: 

373 

374 >>> uri = get_compiled_pattern('^%(URI)s$') 

375 >>> assert uri.match('http://tools.ietf.org/html/rfc3986#appendix-A') 

376 >>> assert not get_compiled_pattern('^%(relative_ref)s$').match('#f#g') 

377 >>> from unicodedata import lookup 

378 >>> smp = 'urn:' + lookup('OLD ITALIC LETTER A') # U+00010300 

379 >>> assert not uri.match(smp) 

380 >>> m = get_compiled_pattern('^%(IRI)s$').match(smp) 

381 

382 On narrow builds, non-BMP characters are (incorrectly) excluded:: 

383 

384 >>> assert NARROW_BUILD == (not m) 

385 

386 For parsing, some subcomponents are captured in named groups (*only if* 

387 regex_ is available, otherwise see `parse`):: 

388 

389 >>> match = uri.match('http://tools.ietf.org/html/rfc3986#appendix-A') 

390 >>> d = match.groupdict() 

391 >>> if REGEX: 

392 ... assert all([ d['scheme'] == 'http', 

393 ... d['authority'] == 'tools.ietf.org', 

394 ... d['path'] == '/html/rfc3986', 

395 ... d['query'] == None, 

396 ... d['fragment'] == 'appendix-A' ]) 

397 

398 >>> for r in patterns.keys(): 

399 ... assert get_compiled_pattern(r) 

400 

401 """ 

402 cache, key = get_compiled_pattern.cache, (rule, flags) 

403 if key not in cache: 

404 if NARROW_BUILD: 

405 pats = bmp_patterns if REGEX else bmp_upatterns_no_names 

406 else: 

407 pats = patterns if REGEX else upatterns_no_names 

408 p = pats.get(rule) or rule % pats 

409 cache[key] = _re.compile(p, flags) 

410 return cache[key] 

411get_compiled_pattern.cache = {}

437def parse(string, rule='IRI_reference'): 

438 """Parses `string` according to `rule` into a dict of subcomponents. 

439 

440 If `rule` is None, parse an IRI_reference `without validation 

441 <http://tools.ietf.org/html/rfc3986#appendix-B>`_. 

442 

443 If regex_ is available, any rule is supported; with re_, `rule` must be 

444 'IRI_reference' or some special case thereof ('IRI', 'absolute_IRI', 

445 'irelative_ref', 'irelative_part', 'URI_reference', 'URI', 'absolute_URI', 

446 'relative_ref', 'relative_part'). :: 

447 

448 >>> d = parse('http://tools.ietf.org/html/rfc3986#appendix-A', 

449 ... rule='URI') 

450 >>> assert all([ d['scheme'] == 'http', 

451 ... d['authority'] == 'tools.ietf.org', 

452 ... d['path'] == '/html/rfc3986', 

453 ... d['query'] == None, 

454 ... d['fragment'] == 'appendix-A' ]) 

455 

456 """ 

457 if not REGEX and rule and rule not in REFERENCE_RULES: 

458 raise ValueError(rule) 

459 if rule: 

460 m = match(string, rule) 

461 if not m: 

462 raise ValueError('%r is not a valid %r.' % (string, rule)) 

463 if REGEX: 

464 return _i2u(m.groupdict()) 

465 return _i2u(_iri_non_validating_re.match(string).groupdict())

468def _i2u(dic): 

469 for (name, iname) in [('authority', 'iauthority'), ('path', 'ipath'), 

470 ('query', 'iquery'), ('fragment', 'ifragment')]: 

471 if dic.get(name) is None: 

472 dic[name] = dic.get(iname) 

473 return dic

476def compose(scheme=None, authority=None, path=None, query=None, fragment=None, 

477 iauthority=None, ipath=None, iquery=None, ifragment=None, **kw): 

478 """Returns an URI composed_ from named parts. 

479 

480 .. _composed: http://tools.ietf.org/html/rfc3986#section-5.3 

481 """ 

482 _i2u(locals()) 

483 res = '' 

484 if scheme is not None: 

485 res += scheme + ':' 

486 if authority is not None: 

487 res += '//' + authority 

488 res += path or '' 

489 if query is not None: 

490 res += '?' + query 

491 if fragment is not None: 

492 res += '#' + fragment 

493 return res

501 c = 1 

502 while c:

504 return path

507def resolve(base, uriref, strict=True, return_parts=False): 

508 """Resolves_ an `URI reference` relative to a `base` URI. 

509 

510 `Test cases <http://tools.ietf.org/html/rfc3986#section-5.4>`_:: 

511 

512 >>> base = resolve.test_cases_base 

513 >>> for relative, resolved in resolve.test_cases.items(): 

514 ... assert resolve(base, relative) == resolved 

515 

516 If `return_parts` is True, returns a dict of named parts instead of 

517 a string. 

518 

519 Examples:: 

520 

521 >>> assert resolve('urn:rootless', '../../name') == 'urn:name' 

522 >>> assert resolve('urn:root/less', '../../name') == 'urn:/name' 

523 >>> assert resolve('http://a/b', 'http:g') == 'http:g' 

524 >>> assert resolve('http://a/b', 'http:g', strict=False) == 'http://a/g' 

525 

526 .. _Resolves: http://tools.ietf.org/html/rfc3986#section-5.2 

527 

528 """ 

529 #base = normalize(base) 

530 if isinstance(base, basestring): 

531 B = parse(base, 'IRI') 

532 else: 

533 B = _i2u(dict(base)) 

534 if not B.get('scheme'): 

535 raise ValueError('Expected an IRI (with scheme), not %r.' % base) 

536 

537 if isinstance(uriref, basestring): 

538 R = parse(uriref, 'IRI_reference') 

539 else: 

540 R = _i2u(dict(uriref)) 

541 

542 # _last_segment = get_compiled_pattern(r'(?<=^|/)%(segment)s$') 

543 

544 if R['scheme'] and (strict or R['scheme'] != B['scheme']): 

545 T = R 

546 else: 

547 T = {} 

548 T['scheme'] = B['scheme'] 

549 if R['authority'] is not None: 

550 T['authority'] = R['authority'] 

551 T['path'] = R['path'] 

552 T['query'] = R['query'] 

553 else: 

554 T['authority'] = B['authority'] 

555 if R['path']: 

556 if R['path'][:1] == "/": 

557 T['path'] = R['path'] 

558 elif B['authority'] is not None and not B['path']: 

559 T['path'] = '/%s' % R['path'] 

560 else: 

561 T['path'] = ''.join(B['path'].rpartition('/')[:2]) + R['path'] 

562 # _last_segment.sub(R['path'], B['path']) 

563 T['query'] = R['query'] 

564 else: 

565 T['path'] = B['path'] 

566 if R['query'] is not None: 

567 T['query'] = R['query'] 

568 else: 

569 T['query'] = B['query'] 

570 T['fragment'] = R['fragment'] 

571 T['path'] = _remove_dot_segments(T['path'])
