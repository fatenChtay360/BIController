68def bind_sockets( 

69 port: int, 

70 address: str = None, 

71 family: socket.AddressFamily = socket.AF_UNSPEC, 

72 backlog: int = _DEFAULT_BACKLOG, 

73 flags: int = None, 

74 reuse_port: bool = False, 

75) -> List[socket.socket]: 

76 """Creates listening sockets bound to the given port and address. 

77 

78 Returns a list of socket objects (multiple sockets are returned if 

79 the given address maps to multiple IP addresses, which is most common 

80 for mixed IPv4 and IPv6 use). 

81 

82 Address may be either an IP address or hostname. If it's a hostname, 

83 the server will listen on all IP addresses associated with the 

84 name. Address may be an empty string or None to listen on all 

85 available interfaces. Family may be set to either `socket.AF_INET` 

86 or `socket.AF_INET6` to restrict to IPv4 or IPv6 addresses, otherwise 

87 both will be used if available. 

88 

89 The ``backlog`` argument has the same meaning as for 

90 `socket.listen() <socket.socket.listen>`. 

91 

92 ``flags`` is a bitmask of AI_* flags to `~socket.getaddrinfo`, like 

93 ``socket.AI_PASSIVE | socket.AI_NUMERICHOST``. 

94 

95 ``reuse_port`` option sets ``SO_REUSEPORT`` option for every socket 

96 in the list. If your platform doesn't support this option ValueError will 

97 be raised. 

98 """ 

99 if reuse_port and not hasattr(socket, "SO_REUSEPORT"): 

100 raise ValueError("the platform doesn't support SO_REUSEPORT") 

101 

102 sockets = [] 

103 if address == "": 

104 address = None 

105 if not socket.has_ipv6 and family == socket.AF_UNSPEC: 

106 # Python can be compiled with --disable-ipv6, which causes 

107 # operations on AF_INET6 sockets to fail, but does not 

108 # automatically exclude those results from getaddrinfo 

109 # results. 

110 # http://bugs.python.org/issue16208 

111 family = socket.AF_INET 

112 if flags is None: 

113 flags = socket.AI_PASSIVE 

114 bound_port = None 

115 unique_addresses = set() # type: set 

116 for res in sorted( 

117 socket.getaddrinfo(address, port, family, socket.SOCK_STREAM, 0, flags), 

118 key=lambda x: x[0], 

119 ): 

120 if res in unique_addresses: 

121 continue 

122 

123 unique_addresses.add(res) 

124 

125 af, socktype, proto, canonname, sockaddr = res 

126 if ( 

127 sys.platform == "darwin" 

128 and address == "localhost" 

129 and af == socket.AF_INET6 

130 and sockaddr[3] != 0 

131 ): 

132 # Mac OS X includes a link-local address fe80::1%lo0 in the 

133 # getaddrinfo results for 'localhost'. However, the firewall 

134 # doesn't understand that this is a local address and will 

135 # prompt for access (often repeatedly, due to an apparent 

136 # bug in its ability to remember granting access to an 

137 # application). Skip these addresses. 

138 continue 

139 try: 

140 sock = socket.socket(af, socktype, proto) 

141 except socket.error as e: 

142 if errno_from_exception(e) == errno.EAFNOSUPPORT: 

143 continue 

144 raise 

145 set_close_exec(sock.fileno()) 

146 if os.name != "nt": 

147 try: 

148 sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) 

149 except socket.error as e: 

150 if errno_from_exception(e) != errno.ENOPROTOOPT: 

151 # Hurd doesn't support SO_REUSEADDR. 

152 raise 

153 if reuse_port: 

154 sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEPORT, 1) 

155 if af == socket.AF_INET6: 

156 # On linux, ipv6 sockets accept ipv4 too by default, 

157 # but this makes it impossible to bind to both 

158 # 0.0.0.0 in ipv4 and :: in ipv6. On other systems, 

159 # separate sockets *must* be used to listen for both ipv4 

160 # and ipv6. For consistency, always disable ipv4 on our 

161 # ipv6 sockets and use a separate ipv4 socket when needed. 

162 # 

163 # Python 2.x on windows doesn't have IPPROTO_IPV6. 

164 if hasattr(socket, "IPPROTO_IPV6"): 

165 sock.setsockopt(socket.IPPROTO_IPV6, socket.IPV6_V6ONLY, 1) 

166 

167 # automatic port allocation with port=None 

168 # should bind on the same port on IPv4 and IPv6 

169 host, requested_port = sockaddr[:2] 

170 if requested_port == 0 and bound_port is not None: 

171 sockaddr = tuple([host, bound_port] + list(sockaddr[2:])) 

172 

173 sock.setblocking(False) 

174 sock.bind(sockaddr) 

175 bound_port = sock.getsockname()[1] 

176 sock.listen(backlog) 

177 sockets.append(sock) 

178 return sockets

220def add_accept_handler(

222) -> Callable[[], None]: 

223 """Adds an `.IOLoop` event handler to accept new connections on ``sock``. 

224 

225 When a connection is accepted, ``callback(connection, address)`` will 

226 be run (``connection`` is a socket object, and ``address`` is the 

227 address of the other end of the connection). Note that this signature 

228 is different from the ``callback(fd, events)`` signature used for 

229 `.IOLoop` handlers. 

230 

231 A callable is returned which, when called, will remove the `.IOLoop` 

232 event handler and stop processing further incoming connections. 

233 

234 .. versionchanged:: 5.0 

235 The ``io_loop`` argument (deprecated since version 4.1) has been removed. 

236 

237 .. versionchanged:: 5.0 

238 A callable is returned (``None`` was returned before). 

239 """ 

240 io_loop = IOLoop.current() 

241 removed = [False]

243 def accept_handler(fd: socket.socket, events: int) -> None: 

244 # More connections may come in while we're handling callbacks; 

245 # to prevent starvation of other tasks we must limit the number 

246 # of connections we accept at a time. Ideally we would accept 

247 # up to the number of connections that were waiting when we 

248 # entered this method, but this information is not available 

249 # (and rearranging this method to call accept() as many times 

250 # as possible before running any callbacks would have adverse 

251 # effects on load balancing in multiprocess configurations). 

252 # Instead, we use the (default) listen backlog as a rough 

253 # heuristic for the number of connections we can reasonably 

254 # accept at once. 

255 for i in range(_DEFAULT_BACKLOG): 

256 if removed[0]: 

257 # The socket was probably closed 

258 return 

259 try: 

260 connection, address = sock.accept() 

261 except socket.error as e: 

262 # _ERRNO_WOULDBLOCK indicate we have accepted every 

263 # connection that is available. 

264 if errno_from_exception(e) in _ERRNO_WOULDBLOCK: 

265 return 

266 # ECONNABORTED indicates that there was a connection 

267 # but it was closed while still in the accept queue. 

268 # (observed on FreeBSD). 

269 if errno_from_exception(e) == errno.ECONNABORTED: 

270 continue 

271 raise 

272 set_close_exec(connection.fileno()) 

273 callback(connection, address)

275 def remove_handler() -> None: 

276 io_loop.remove_handler(sock) 

277 removed[0] = True 

278 

279 io_loop.add_handler(sock, accept_handler, IOLoop.READ) 

280 return remove_handler

371def _resolve_addr( 

372 host: str, port: int, family: socket.AddressFamily = socket.AF_UNSPEC 

373) -> List[Tuple[int, Any]]: 

374 # On Solaris, getaddrinfo fails if the given port is not found 

375 # in /etc/services and no socket type is given, so we must pass 

376 # one here. The socket type used here doesn't seem to actually 

377 # matter (we discard the one we get back in the results), 

378 # so the addresses we return should still be usable with SOCK_DGRAM. 

379 addrinfo = socket.getaddrinfo(host, port, family, socket.SOCK_STREAM) 

380 results = [] 

381 for fam, socktype, proto, canonname, address in addrinfo: 

382 results.append((fam, address)) 

383 return results

457class ThreadedResolver(ExecutorResolver): 

458 """Multithreaded non-blocking `Resolver` implementation. 

459 

460 Requires the `concurrent.futures` package to be installed 

461 (available in the standard library since Python 3.2, 

462 installable with ``pip install futures`` in older versions). 

463

471 size is set by the first one to be created. 

472

474 The default `Resolver` now uses `.IOLoop.run_in_executor`; use that instead

481 def initialize(self, num_threads: int = 10) -> None: # type: ignore

483 super(ThreadedResolver, self).initialize(

487 @classmethod 

488 def _create_threadpool( 

489 cls, num_threads: int 

490 ) -> concurrent.futures.ThreadPoolExecutor:

502class OverrideResolver(Resolver): 

503 """Wraps a resolver with a mapping of overrides. 

504 

505 This can be used to make local DNS changes (e.g. for testing) 

506 without modifying system-wide settings. 

507 

508 The mapping can be in three formats:: 

509 

510 { 

511 # Hostname to host or ip 

512 "example.com": "127.0.1.1", 

513 

514 # Host+port to host+port 

515 ("login.example.com", 443): ("localhost", 1443), 

516 

517 # Host+port+address family to host+port 

518 ("login.example.com", 443, socket.AF_INET6): ("::1", 1443), 

519 } 

520 

521 .. versionchanged:: 5.0 

522 Added support for host-port-family triplets. 

523 """ 

524 

525 def initialize(self, resolver: Resolver, mapping: dict) -> None: 

526 self.resolver = resolver 

527 self.mapping = mapping 

528 

529 def close(self) -> None: 

530 self.resolver.close() 

531 

532 def resolve( 

533 self, host: str, port: int, family: socket.AddressFamily = socket.AF_UNSPEC 

534 ) -> Awaitable[List[Tuple[int, Any]]]: 

535 if (host, port, family) in self.mapping: 

536 host, port = self.mapping[(host, port, family)] 

537 elif (host, port) in self.mapping: 

538 host, port = self.mapping[(host, port)] 

539 elif host in self.mapping: 

540 host = self.mapping[host] 

541 return self.resolver.resolve(host, port, family)

552def ssl_options_to_context( 

553 ssl_options: Union[Dict[str, Any], ssl.SSLContext] 

554) -> ssl.SSLContext: 

555 """Try to convert an ``ssl_options`` dictionary to an 

556 `~ssl.SSLContext` object. 

557 

558 The ``ssl_options`` dictionary contains keywords to be passed to 

559 `ssl.wrap_socket`. In Python 2.7.9+, `ssl.SSLContext` objects can 

560 be used instead. This function converts the dict form to its 

561 `~ssl.SSLContext` equivalent, and may be used when a component which 

562 accepts both forms needs to upgrade to the `~ssl.SSLContext` version 

563 to use features like SNI or NPN. 

564 """ 

565 if isinstance(ssl_options, ssl.SSLContext): 

566 return ssl_options 

567 assert isinstance(ssl_options, dict) 

568 assert all(k in _SSL_CONTEXT_KEYWORDS for k in ssl_options), ssl_options 

569 # Can't use create_default_context since this interface doesn't 

570 # tell us client vs server. 

571 context = ssl.SSLContext(ssl_options.get("ssl_version", ssl.PROTOCOL_SSLv23)) 

572 if "certfile" in ssl_options: 

573 context.load_cert_chain( 

574 ssl_options["certfile"], ssl_options.get("keyfile", None) 

575 ) 

576 if "cert_reqs" in ssl_options: 

577 context.verify_mode = ssl_options["cert_reqs"] 

578 if "ca_certs" in ssl_options: 

579 context.load_verify_locations(ssl_options["ca_certs"]) 

580 if "ciphers" in ssl_options: 

581 context.set_ciphers(ssl_options["ciphers"]) 

582 if hasattr(ssl, "OP_NO_COMPRESSION"): 

583 # Disable TLS compression to avoid CRIME and related attacks. 

584 # This constant depends on openssl version 1.0. 

585 # TODO: Do we need to do this ourselves or can we trust 

586 # the defaults? 

587 context.options |= ssl.OP_NO_COMPRESSION 

588 return context
