46class HTTPServer(TCPServer, Configurable, httputil.HTTPServerConnectionDelegate): 

47 r"""A non-blocking, single-threaded HTTP server. 

48 

49 A server is defined by a subclass of `.HTTPServerConnectionDelegate`, 

50 or, for backwards compatibility, a callback that takes an 

51 `.HTTPServerRequest` as an argument. The delegate is usually a 

52 `tornado.web.Application`. 

53 

54 `HTTPServer` supports keep-alive connections by default 

55 (automatically for HTTP/1.1, or for HTTP/1.0 when the client 

56 requests ``Connection: keep-alive``). 

57 

58 If ``xheaders`` is ``True``, we support the 

59 ``X-Real-Ip``/``X-Forwarded-For`` and 

60 ``X-Scheme``/``X-Forwarded-Proto`` headers, which override the 

61 remote IP and URI scheme/protocol for all requests. These headers 

62 are useful when running Tornado behind a reverse proxy or load 

63 balancer. The ``protocol`` argument can also be set to ``https`` 

64 if Tornado is run behind an SSL-decoding proxy that does not set one of 

65 the supported ``xheaders``. 

66 

67 By default, when parsing the ``X-Forwarded-For`` header, Tornado will 

68 select the last (i.e., the closest) address on the list of hosts as the 

69 remote host IP address. To select the next server in the chain, a list of 

70 trusted downstream hosts may be passed as the ``trusted_downstream`` 

71 argument. These hosts will be skipped when parsing the ``X-Forwarded-For`` 

72 header. 

73 

74 To make this server serve SSL traffic, send the ``ssl_options`` keyword 

75 argument with an `ssl.SSLContext` object. For compatibility with older 

76 versions of Python ``ssl_options`` may also be a dictionary of keyword 

77 arguments for the `ssl.wrap_socket` method.:: 

78 

79 ssl_ctx = ssl.create_default_context(ssl.Purpose.CLIENT_AUTH) 

80 ssl_ctx.load_cert_chain(os.path.join(data_dir, "mydomain.crt"), 

81 os.path.join(data_dir, "mydomain.key")) 

82 HTTPServer(application, ssl_options=ssl_ctx) 

83 

84 `HTTPServer` initialization follows one of three patterns (the 

85 initialization methods are defined on `tornado.tcpserver.TCPServer`): 

86 

87 1. `~tornado.tcpserver.TCPServer.listen`: simple single-process:: 

88 

89 server = HTTPServer(app) 

90 server.listen(8888) 

91 IOLoop.current().start() 

92 

93 In many cases, `tornado.web.Application.listen` can be used to avoid 

94 the need to explicitly create the `HTTPServer`. 

95 

96 2. `~tornado.tcpserver.TCPServer.bind`/`~tornado.tcpserver.TCPServer.start`: 

97 simple multi-process:: 

98 

99 server = HTTPServer(app) 

100 server.bind(8888) 

101 server.start(0) # Forks multiple sub-processes 

102 IOLoop.current().start() 

103 

104 When using this interface, an `.IOLoop` must *not* be passed 

105 to the `HTTPServer` constructor. `~.TCPServer.start` will always start 

106 the server on the default singleton `.IOLoop`. 

107 

108 3. `~tornado.tcpserver.TCPServer.add_sockets`: advanced multi-process:: 

109 

110 sockets = tornado.netutil.bind_sockets(8888) 

111 tornado.process.fork_processes(0) 

112 server = HTTPServer(app) 

113 server.add_sockets(sockets) 

114 IOLoop.current().start() 

115 

116 The `~.TCPServer.add_sockets` interface is more complicated, 

117 but it can be used with `tornado.process.fork_processes` to 

118 give you more flexibility in when the fork happens. 

119 `~.TCPServer.add_sockets` can also be used in single-process 

120 servers if you want to create your listening sockets in some 

121 way other than `tornado.netutil.bind_sockets`. 

122 

123 .. versionchanged:: 4.0 

124 Added ``decompress_request``, ``chunk_size``, ``max_header_size``, 

125 ``idle_connection_timeout``, ``body_timeout``, ``max_body_size`` 

126 arguments. Added support for `.HTTPServerConnectionDelegate` 

127 instances as ``request_callback``. 

128 

129 .. versionchanged:: 4.1 

130 `.HTTPServerConnectionDelegate.start_request` is now called with 

131 two arguments ``(server_conn, request_conn)`` (in accordance with the 

132 documentation) instead of one ``(request_conn)``. 

133 

134 .. versionchanged:: 4.2 

135 `HTTPServer` is now a subclass of `tornado.util.Configurable`. 

136 

137 .. versionchanged:: 4.5 

138 Added the ``trusted_downstream`` argument. 

139 

140 .. versionchanged:: 5.0 

141 The ``io_loop`` argument has been removed. 

142 """

204 async def close_all_connections(self) -> None: 

205 """Close all open connections and asynchronously wait for them to finish. 

206 

207 This method is used in combination with `~.TCPServer.stop` to 

208 support clean shutdowns (especially for unittests). Typical 

209 usage would call ``stop()`` first to stop accepting new 

210 connections, then ``await close_all_connections()`` to wait for 

211 existing connections to finish. 

212 

213 This method does not currently close open websocket connections. 

214 

215 Note that this method is a coroutine and must be caled with ``await``. 

216 

217 """ 

218 while self._connections: 

219 # Peek at an arbitrary element of the set 

220 conn = next(iter(self._connections)) 

221 await conn.close()

286class _HTTPRequestContext(object): 

287 def __init__( 

288 self, 

289 stream: iostream.IOStream, 

290 address: Tuple, 

291 protocol: Optional[str], 

292 trusted_downstream: List[str] = None, 

293 ) -> None: 

294 self.address = address 

295 # Save the socket's address family now so we know how to 

296 # interpret self.address even after the stream is closed 

297 # and its socket attribute replaced with None. 

298 if stream.socket is not None: 

299 self.address_family = stream.socket.family 

300 else: 

301 self.address_family = None 

302 # In HTTPServerRequest we want an IP, not a full socket address. 

303 if ( 

304 self.address_family in (socket.AF_INET, socket.AF_INET6) 

305 and address is not None 

306 ): 

307 self.remote_ip = address[0] 

308 else: 

309 # Unix (or other) socket; fake the remote address. 

310 self.remote_ip = "0.0.0.0" 

311 if protocol: 

312 self.protocol = protocol 

313 elif isinstance(stream, iostream.SSLIOStream): 

314 self.protocol = "https" 

315 else: 

316 self.protocol = "http" 

317 self._orig_remote_ip = self.remote_ip 

318 self._orig_protocol = self.protocol 

319 self.trusted_downstream = set(trusted_downstream or []) 

320 

321 def __str__(self) -> str: 

322 if self.address_family in (socket.AF_INET, socket.AF_INET6): 

323 return self.remote_ip 

324 elif isinstance(self.address, bytes): 

325 # Python 3 with the -bb option warns about str(bytes), 

326 # so convert it explicitly. 

327 # Unix socket addresses are str on mac but bytes on linux. 

328 return native_str(self.address) 

329 else: 

330 return str(self.address)

332 def _apply_xheaders(self, headers: httputil.HTTPHeaders) -> None: 

333 """Rewrite the ``remote_ip`` and ``protocol`` fields.""" 

334 # Squid uses X-Forwarded-For, others use X-Real-Ip 

335 ip = headers.get("X-Forwarded-For", self.remote_ip) 

336 # Skip trusted downstream hosts in X-Forwarded-For list 

337 for ip in (cand.strip() for cand in reversed(ip.split(","))): 

338 if ip not in self.trusted_downstream: 

339 break 

340 ip = headers.get("X-Real-Ip", ip) 

341 if netutil.is_valid_ip(ip): 

342 self.remote_ip = ip 

343 # AWS uses X-Forwarded-Proto 

344 proto_header = headers.get( 

345 "X-Scheme", headers.get("X-Forwarded-Proto", self.protocol) 

346 ) 

347 if proto_header: 

348 # use only the last proto entry if there is more than one 

349 # TODO: support trusting mutiple layers of proxied protocol 

350 proto_header = proto_header.split(",")[-1].strip() 

351 if proto_header in ("http", "https"): 

352 self.protocol = proto_header 

353 

354 def _unapply_xheaders(self) -> None: 

355 """Undo changes from `_apply_xheaders`. 

356 

357 Xheaders are per-request so they should not leak to the next 

358 request on the same connection. 

359 """ 

360 self.remote_ip = self._orig_remote_ip 

361 self.protocol = self._orig_protocol
